let blue = 0
let green = 0
let red = 0
let strip = false
let item = neopixel.create(DigitalPin.P15, 4, NeoPixelMode.RGB)
basic.forever(function () {
    if (maqueen.sensor(PingUnit.Centimeters) < 35 && maqueen.sensor(PingUnit.Centimeters) != 0) {
        strip = Math.randomBoolean()
        if (strip == true) {
            maqueen.MotorRun(maqueen.aMotors.M1, maqueen.Dir.CW, 110)
            maqueen.MotorRun(maqueen.aMotors.M2, maqueen.Dir.CCW, 110)
            basic.pause(800)
        }
        if (strip == false) {
            maqueen.MotorRun(maqueen.aMotors.M1, maqueen.Dir.CCW, 110)
            maqueen.MotorRun(maqueen.aMotors.M2, maqueen.Dir.CW, 110)
            basic.pause(800)
        }
    } else {
        maqueen.MotorRun(maqueen.aMotors.M1, maqueen.Dir.CW, 255)
        maqueen.MotorRun(maqueen.aMotors.M2, maqueen.Dir.CW, 255)
    }
})
basic.forever(function () {
    maqueen.writeLED(maqueen.LED.LEDLeft, maqueen.LEDswitch.turnOn)
    maqueen.writeLED(maqueen.LED.LEDRight, maqueen.LEDswitch.turnOn)
})
basic.forever(function () {
    red = 0
    green = 0
    blue = 255
    for (let index = 0; index < 255; index++) {
        red += 1
        blue += -1
        item.showColor(neopixel.rgb(red, green, blue))
        basic.pause(1)
    }
    for (let index = 0; index < 4; index++) {
        for (let index = 0; index < 255; index++) {
            green += 1
            red += -1
            item.showColor(neopixel.rgb(red, green, blue))
            basic.pause(1)
        }
    }
})
basic.forever(function () {
    basic.showIcon(IconNames.Heart)
})
